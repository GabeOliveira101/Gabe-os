        // HELPERS

        function getRandomQuote() {

            return quotes[Math.floor(Math.random() * quotes.length)];

        }


        function saveQuote(quote) {

            localStorage.setItem('lastShownQuote', JSON.stringify(quote));

        }


        function loadQuote() {

            const saved = localStorage.getItem('lastShownQuote');

            return saved ? JSON.parse(saved) : getRandomQuote();

        }


        function saveFocus(text) {

            localStorage.setItem('todayFocus', text);

        }


        function loadFocus() {

            return localStorage.getItem('todayFocus') || '';

        }


        function addWin(text) {

            const wins = loadWins();

            const timestamp = new Date().toISOString();

            wins.unshift({ text, timestamp });

            localStorage.setItem('wins', JSON.stringify(wins));

            return wins;

        }


        function loadWins() {

            const saved = localStorage.getItem('wins');

            return saved ? JSON.parse(saved) : [];

        }


        function formatTimestamp(isoString) {

            const d = new Date(isoString);

            const month = String(d.getMonth() + 1).padStart(2, '0');

            const day = String(d.getDate()).padStart(2, '0');

            const year = d.getFullYear();

            const hours = String(d.getHours()).padStart(2, '0');

            const minutes = String(d.getMinutes()).padStart(2, '0');

            return `${year}-${month}-${day} ${hours}:${minutes}`;

        }


        // INIT DATE/GREETING

        function initDateGreeting() {

            const now = new Date();

            const options = { month: 'short', day: 'numeric', year: 'numeric' };

            const dateStr = now.toLocaleDateString('en-US', options).toUpperCase().replace(',', '');

            document.getElementById('dateGreeting').textContent = `${dateStr} — STAY DANGEROUS, GABE`;

        }


        // INIT QUOTE

        function initQuote() {

            const quote = loadQuote();

            displayQuote(quote);

        }


        function displayQuote(quote) {

            document.getElementById('quoteText').textContent = `"${quote.text}"`;

            document.getElementById('quoteAuthor').textContent = `— ${quote.author}`;

        }


        // INIT FOCUS

        function initFocus() {

            const focusInput = document.getElementById('focusInput');

            focusInput.value = loadFocus();

            

            focusInput.addEventListener('input', (e) => {

                saveFocus(e.target.value);

            });

        }


        // INIT WINS

        function initWins() {

            renderWins();

            

            document.getElementById('addWinBtn').addEventListener('click', () => {

                document.getElementById('winInputContainer').classList.add('active');

                document.getElementById('winInput').focus();

            });


            document.getElementById('winCancelBtn').addEventListener('click', () => {

                document.getElementById('winInputContainer').classList.remove('active');

                document.getElementById('winInput').value = '';

            });


            document.getElementById('winSubmitBtn').addEventListener('click', submitWin);

            

            document.getElementById('winInput').addEventListener('keypress', (e) => {

                if (e.key === 'Enter') submitWin();

            });

        }


        function submitWin() {

            const input = document.getElementById('winInput');

            const text = input.value.trim();

            

            if (!text) return;

            

            addWin(text);

            renderWins();

            

            input.value = '';

            document.getElementById('winInputContainer').classList.remove('active');

        }


        function renderWins() {

            const wins = loadWins();

            const list = document.getElementById('winsList');

            

            if (wins.length === 0) {

                list.innerHTML = '<div class="empty-state">NO WINS LOGGED YET. GET AFTER IT.</div>';

                return;

            }

            

            list.innerHTML = wins.map(win => `

                <li class="win-item">

                    <div class="win-text">${win.text}</div>

                    <div class="win-timestamp">${formatTimestamp(win.timestamp)}</div>

                </li>

            `).join('');

        }

        document.getElementById('newQuoteBtn').addEventListener('click', () => {

            displayQuote(newQuote);

        });
